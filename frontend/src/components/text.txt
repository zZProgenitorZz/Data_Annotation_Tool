            name: newData || newData.name || "Untitled Dataset",

            completed_Images: 0,

            location_of_collection: newData.location_of_collection || "N/A",
            description: newData.description || "",
          };
          setDatasetsState((prev) => [formatted, ...prev]);

        }}
